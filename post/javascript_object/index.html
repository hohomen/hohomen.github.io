객체(JavaScript Basic)

February 20, 2019     ""

3 minute read




1.개요
#배열(Javascript에서 자료형은 객체이다.)은 객체를 기반으로 만들어졌으므로 배열과 객체는 상당히 비슷합니다. 
다른 점이 있다면 
배열은 요소에 접근할 때 인덱스를 사용하지만
var array = ['사과', '바나나', '망고', '딸기']; (배열 선언)

객체는 키를 사용한다는 점입니다.
var product = {
            제품명: '7D 건조 망고',
            유형: '당절임',
            성분: '망고, 설탕, 메타중아황산나트륨, 치자황색소',
            원산지: '필리핀'
        };

#요소에 접근하는 방법은 2가지가 있습니다.
1. 배열과 비슷한 방법
product['제품명']
product['유형']
product['성분']
product['원산지']
2. 일반적으로 사용하는 방법
product.제품명
product.유형
product.성분
product.원산지

#객체의 키는 식별자 또는 문자열을 모두 사용할 수 있습니다.
var object = {
            'with space': 273,
            'with ~!@# $%^&*()_+': 52
        };
식별자가 아닌 문자를 키로 사용했을 때는 무조건 대괄호를 사용해야 객체의 요소에 접근할 수 있습니다. object['with space'],  object['with ~!@# $%^&*()_+']



2.속성과 메서드
배열 내부에 있는 값을 요소(element) (인덱스 - 요소)
객체 내부에 있는 값은 속성(property) (키 - 속성)라고 합니다.
함수를 메서드라고 부릅니다.

#(this) 메서드 내에서 자기 자신이 속성을 출력하고 싶을 때는 자신이 가진 속성임을 분명하게 표시해야 합니다.
C++, 자바과 같은 프로그래밍 언어는 같은 객체 내부에서 this 키워드를 생략할 수 있지만 자바스크립트는 그렇지 못합니다.
(this = 객체 자기 자신)

var person = {
            name: '윤인성',
            eat: function (food) {
                alert(this.name + '이 ' + food + '을/를 먹습니다.');
            }
        };
        // 메서드를 호출합니다.
        person.eat('밥');
}



3. 객체와 반복문
객체의 속성을 모두 살펴보려면 for in 반복문을 사용해야 합니다.
(배열은 for, for in 모두 사용 가능)

var product = {
            name: 'Microsoft Visual Studio 2015 Ultimate',
            price: '15,000,000원',
            language: '한국어',
            supportOS: 'Win 32/64',
            subscription: true
        };
        // 출력합니다.
        var output = '';
        for (var key in product) {
            output += '●' + key + ': ' + product[key] + '\n';
        }
        alert(output);

Cf. java) for each문.(오직 배열의 값만 읽을(read) 수 있고 수정(write)할 수는 없다.
for(변수타입 변수이름 : 배열이름)



4. in, with 키워드
#in 키워드를 for 키워드와 별도로 사용하면 해당 *키가 객체 안에 있는 지 확인할 수 있습니다.

var output = '';
        var student = {
            이름: '연하진',
            국어: 92, 수학: 98,
            영어: 96, 과학: 98
        };
    output += "'이름' in student: " + ('이름' in student) + '\n';
         output += "'성별' in student: " + ('성별' in student);
            alert(output);

//결과 '이름' in student true
         '이름' in student false

# with 키워는 복잡하게 사용해야 하는 코드를 짧게 줄여주는 키워드입니다.

with(<객체>){
    <코드>
}

-Before
var output = '';
        output += '이름: ' + student.이름 + '\n';
        output += '국어: ' + student.국어 + '\n';
        output += '수학: ' + student.수학 + '\n';
        output += '영어: ' + student.영어 + '\n';
        output += '과학: ' + student.과학 + '\n';
        output += '총점: ' + (student.국어 + student.수학 + student.영어 + student.과학);
        alert(output);

-After
with (student) {
            output += '이름: ' + 이름 + '\n';
            output += '국어: ' + 국어 + '\n';
            output += '수학: ' + 수학 + '\n';
            output += '영어: ' + 영어 + '\n';
            output += '과학: ' + 과학 + '\n';
            output += '총점: ' + (국어 + 수학 + 영어 + 과학);
        }
        alert(output);
        


5. 속성 추가, 제거
var student = {};
추가
student.이름 = '윤인성';
student.취미 = '악기';
student.특기 = '프로그래밍';
student.장래희망 = '생명공학자';
삭제
delete(student.장래희망);



6. 배열에 데이터 추가
객체가 흩어져 있으면 조작이 어려우므로 배열 내에 저장합니다. 배열의 push() 메서드를 사용하면 배열에 요소를 넣을 수 있습니다.
    var students = [];
        // 배열에 요소를 넣습니다.
        students.push({ 이름: '윤인성', 국어: 87, 수학: 98, 영어: 88, 과학: 95 });
        students.push({ 이름: '연하진', 국어: 92, 수학: 98, 영어: 96, 과학: 98 });
        students.push({ 이름: '구지연', 국어: 76, 수학: 96, 영어: 94, 과학: 90 });
        students.push({ 이름: '나선주', 국어: 98, 수학: 92, 영어: 96, 과학: 92 });
